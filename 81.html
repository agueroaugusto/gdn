SQL Statement:SELECT a."CODIGOPEDIDO", MAX(a."FECHAPEDIDO") as FECHAPEDIDO, MAX(a."FECHAENTREGA") as FECHAENTREGA, MAX(b."FIN_STATUS_NBR") as FIN_STATUS_NBR, MAX(b."STATUS_DATE") as STATUS_DATE, CASE WHEN MAX(b."FIN_STATUS_NBR") = 40 AND MAX(b."STATUS_DATE") <= MAX(a."FECHAENTREGA") THEN 'VERDE' WHEN MAX(b."FIN_STATUS_NBR") = 30 AND MAX(b."STATUS_DATE") <= MAX(a."FECHAENTREGA") THEN 'AMARILLO' ELSE 'ROJO' END AS "SEMAFORO" FROM DWH.ECOMMERCE.STG_ORIG_ORDER_CABECERA a INNER JOIN DWH.ECOMMERCE.STG_STAT_ORDER_DETALLE b ON a."CODIGOPEDIDO" = b."CODIGOPEDIDO" WHERE b."FIN_STATUS_NBR" IN (30, 40) GROUP BY a."CODIGOPEDIDO"

SQL Statement:/* V_ULTIMA_FECHA_SEMANAS_ATRAS */ with cte_fin_anterior as ( select fecha as fin_semana_anterior from dwh.datosmaestros.lu_fechas where fecha_dia_nombre = 'Sabado' ) select fecha, fin_semana_anterior as fecha_1, dateadd(day, -7, fin_semana_anterior) as fecha_2, dateadd(day, -14, fin_semana_anterior) as fecha_3, dateadd(day, -21, fin_semana_anterior) as fecha_4 from ( select f.fecha, cfa.fin_semana_anterior , row_number() over (partition by f.fecha order by fin_semana_anterior desc) as rnum from dwh.datosmaestros.lu_fechas f left join cte_fin_anterior cfa on f.fecha> fin_semana_anterior ) where rnum = 1
