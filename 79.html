SQL Statement:/* V_FIRST_ACTIVE_FUTURE_COST_RMS */ with first_active_date as ( --Se regresan los datos que tengan la menor active date que sea superior a la fecha de la foto del dato por cada combinación SKU/Location --En caso de que todos los active date sean menores a la fecha de la foto, entonces se regresa el último active date -- El resultado será el valor de fecha "válido" para cada combinación select sku, location, cast(fecha_carga as date) as fecha_carga, coalesce(min(case when active_date>fecha_carga then active_date end), max(case when active_date<=fecha_carga then active_date end)) as first_active from DWH.CONTROL_INTEGRIDAD.FUTURE_COST_RMS group by sku, location, cast(fecha_carga as date) ), first_active_date_actualizaciones as ( -- se tienen los valores de future cost para cada día de actualización, pero como lo queremos pasar a foto diaria del último precio válido, hay que convertirlo select fcr.sku, fcr.supplier, fcr.location, fcr.base_cost, fcr.net_cost, fcr.net_net_cost, fcr.calc_date, fcr.active_date, fcr.fecha_carga from DWH.CONTROL_INTEGRIDAD.FUTURE_COST_RMS fcr join first_active_date on first_active_date.sku = fcr.sku and first_active_date.location = fcr.location and first_active_date.fecha_carga = cast(fcr.fecha_carga as date) and first_active_date.first_active = fcr.active_date ), fechas_L24M as ( --se genera el calendario para el periodo select fecha from dwh.datosmaestros.lu_fechas where fecha between dateadd(year,-2,current_date-1) and current_date ), fanout_por_fechas as ( -- por cada fecha, regresa cual fue la última actualización por cada sku.location select fecha,sku,location,fecha_carga,first_active from ( select *, rank () over (partition by fecha,sku,location order by fecha_carga desc) r from fechas_L24M fl24m join first_active_date fad on fl24m.fecha >= fad.fecha_carga ) rnk where r=1 ) -- se abre el anterior que era por actualizaciones a una foto diaria select fada.sku, fada.supplier, fada.location, fada.base_cost, fada.net_cost, fada.net_net_cost, fada.calc_date, active_date, fpf.fecha as fecha_carga from first_active_date_actualizaciones fada join fanout_por_fechas fpf on fpf.sku = fada.sku and fpf.location = fada.location and fpf.fecha_carga = cast(fada.fecha_carga as date) and fpf.first_active = fada.active_date
